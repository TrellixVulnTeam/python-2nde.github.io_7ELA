Direction d’une droite

Dans le programme ci-dessous :

  • À chaque passage dans la boucle, la tortue effectue un déplacement horizontal puis un déplacement vertical, jusqu'à atteindre le bord de la fenêtre.
  • Une ligne reliant les points de départ des déplacements horizontaux est tracée en rouge.

Les carreaux ont pour côtés 10 unités.

import turtle
turtle.set_defaults(
    canvwidth = 600,
    canvheight = 600,
    turtle_canvas_wrapper = document['turtle-div']
   # , turtle_canvas_id = 'turtle-canvas' # default, could be overriden
  )
# Espace de jeu
taille = 300
x_pos_axe, y_pos_axe = 0, 0
x_max = 600

def deplacer(t, x, y):
    t.goto(x-300+x_pos_axe, y-300+y_pos_axe)

# Grille
t = turtle.Turtle()
t.speed(0)
t.hideturtle()
t.width(1)
t.color('yellow')

def create_line(x0, y0, x1, y1):
    t.penup()
    t.goto(x0, y0)
    t.pendown()
    t.goto(x1, y1)

i = -taille
while i <= taille:
    create_line(-taille, i, taille, i)
    create_line(i, -taille, i, taille)
    i = i + 10

# Mise en place des tortues
tortue_cible = turtle.Turtle()
tortue_cible.shape("circle")
tortue_cible.color("lightgreen")
tortue_cible.penup()
tortue_cible.speed(0)
tortue_cible.goto(0, 90)

tortue = turtle.Turtle()
tortue.shape("turtle")
tortue.penup()
tortue.speed(0)
tortue.goto(-290, -280)
tortue.pendown()

point_rouge = turtle.Turtle()
point_rouge.pencolor("red")
point_rouge.penup()
point_rouge.speed(0)
point_rouge.goto(-290, -280)
point_rouge.dot(5, "red")
point_rouge.pendown()

def marche(x, y):
    tortue.forward(x)
    tortue.left(90)
    tortue.forward(y)
    tortue.right(90)

# Vitesse de 1 à 50, 0 pour le plus vite possible
tortue.speed(8)
point_rouge.speed(0)

# Initialisation des coordonnées de la tortue
xT = 10
yT = 20

turtle.done()

# Valeurs à changer
deplacement_horizontal = 10
deplacement_vertical = 10

# Déplacement
while xT < x_max:
    xT = xT + deplacement_horizontal
    deplacer(tortue, xT, yT)
    yT = yT + deplacement_vertical
    deplacer(tortue, xT, yT)

# Tracé de la ligne rouge
deplacer(point_rouge, xT, yT)

Exercice
