Atteindre la cible

Dans le programme ci-dessous :

  • À chaque passage dans la boucle, la tortue dessine une marche d’un escalier, jusqu'à atteindre le bord de la fenêtre.
  • Une ligne reliant le point de départ aux points où la tortue tourne à droite est dessinée en rouge.
import turtle
turtle.set_defaults(
    canvwidth = 600,
    canvheight = 600,
    turtle_canvas_wrapper = document['turtle-div']
   # , turtle_canvas_id = 'turtle-canvas' # default, could be overriden
  )
# Espace de jeu
taille = 300

def visible(tortue):
    return tortue.xcor() >= -taille and \
           tortue.xcor() <=  taille and \
           tortue.ycor() >= -taille and \
           tortue.ycor() <=  taille

def lancer_animation():
    turtle.done()

# Grille
t = turtle.Turtle()
t.speed(0)
t.hideturtle()
t.width(1)
t.color('yellow')

def create_line(x0, y0, x1, y1):
    t.penup()
    t.goto(x0, y0)
    t.pendown()
    t.goto(x1, y1)

i = -taille
while i <= taille:
    create_line(-taille, i, taille, i)
    create_line(i, -taille, i, taille)
    i = i + 10

# Mise en place des tortues
tortue_cible = turtle.Turtle()
tortue_cible.shapesize(5)
tortue_cible.shape("circle")
tortue_cible.color("lightgreen")
tortue_cible.penup()
tortue_cible.speed(0)
tortue_cible.goto(0, 90)

tortue = turtle.Turtle()
tortue.shape("turtle")
tortue.penup()
tortue.speed(0)
tortue.goto(-290, -280)
tortue.pendown()

point_rouge = turtle.Turtle()
point_rouge.pencolor("red")
point_rouge.penup()
point_rouge.speed(0)
point_rouge.goto(-290, -280)
point_rouge.dot(5, "red")
point_rouge.pendown()

# Vitesse de 1 à 50, 0 pour le plus vite possible
tortue.speed(8)
point_rouge.speed(0)

turtle.done()

# Valeurs à changer
deplacement_horizontal = 10
deplacement_vertical = 10

# Déplacement
while visible(tortue):
    tortue.forward(deplacement_horizontal)
    tortue.left(90)
    tortue.forward(deplacement_vertical)
    tortue.right(90)

point_rouge.goto(tortue.position())

lancer_animation()
