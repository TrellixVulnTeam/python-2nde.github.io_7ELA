Direction d’une droite (suite)

import turtle
turtle.set_defaults(
    canvwidth = 600,
    canvheight = 600,
    turtle_canvas_wrapper = document['turtle-div']
   # , turtle_canvas_id = 'turtle-canvas' # default, could be overriden
  )
# Espace de jeu
taille = 300
x_pos_axe, y_pos_axe = 200, 200

def visible(tortue):
    return tortue.xcor() >= -taille and \
           tortue.xcor() <=  taille and \
           tortue.ycor() >= -taille and \
           tortue.ycor() <=  taille

def deplacer(t, x, y):
    t.goto(x-300+x_pos_axe, y-300+y_pos_axe)

def A(x, y):
    deplacer(tortue, x, y)
    deplacer(point_rouge, x, y)
    point_rouge.dot(5, "red")
    point_rouge.write("   A")

def B(x, y):
    deplacer(t, x, y-10)
    t.write("  B")
    deplacer(t, x, y)
    t.dot(10, "lightgreen")
    t.dot(1, "black")

# Grille
t = turtle.Turtle()
t.speed(0)
t.hideturtle()
t.width(1)
t.color('yellow')

def create_line(x0, y0, x1, y1):
    t.penup()
    t.goto(x0, y0)
    t.pendown()
    t.goto(x1, y1)

i = -taille
while i <= taille:
    create_line(-taille, i, taille, i)
    create_line(i, -taille, i, taille)
    i = i + 10

# Repere
t.color('black')
# Axes
create_line(-taille, -taille + y_pos_axe, taille, -taille + y_pos_axe)
create_line(-taille + x_pos_axe, -taille, -taille + x_pos_axe, taille)
# Tirets
for i in range(-1, 4):
    create_line(i*100 - 100, -5 - 100, i*100 - 100, 5 - 100)
    create_line(-5 - 100, i*100 - 100, 5 - 100, i*100 - 100)
# Nom des axes et échelle
t.penup()
deplacer(t, 380, 5)
t.write("x")
deplacer(t, 5, 380)
t.write("y")
deplacer(t, 85, -25)
t.write("100")
deplacer(t, -45, 95)
t.write("100")

# Mise en place des tortues
tortue = turtle.Turtle()
tortue.shape("turtle")
tortue.penup()
tortue.speed(0)

point_rouge = turtle.Turtle()
point_rouge.pencolor("red")
point_rouge.penup()
point_rouge.speed(0)

A(10, 20)
B(300, 390)

def marche(x, y):
    tortue.forward(x)
    tortue.left(90)
    tortue.forward(y)
    tortue.right(90)

# Vitesse de 1 à 50, 0 pour le plus vite possible
tortue.speed(8)
point_rouge.speed(0)
tortue.pendown()
point_rouge.pendown()

turtle.done()

point_rouge.goto(tortue.position())

A(10, 20)
B(300, 390)

# Valeurs à changer
Δx = 10
Δy = 10

# Déplacement
while visible(tortue):
    marche(Δx, Δy)
