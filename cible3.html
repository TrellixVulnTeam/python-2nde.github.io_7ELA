Exercice des quatre cibles

Le point de départ est noté A et a pour coordonnées (10, 20).

Les quatre cibles sont nommées B, C, D et E dans cet ordre et en partant du haut. Elles ont pour coordonnées :

  • B(250, 390)
  • C(250, 200)
  • D(250, 20)
  • E(250, -100)
taille = 300  # moitié du côté du playground

turtle.set_defaults(
    canvwidth = 600,
    canvheight = 600,
    turtle_canvas_wrapper = document['turtle-div']
   # , turtle_canvas_id = 'turtle-canvas' # default, could be overriden
  )

# Espace de jeu
x_pos_axe, y_pos_axe = 200, 200
x_max = 2*taille

font = "Arial", "30px", "normal"

def deplacer(t, x, y):
    t.goto(x-taille+x_pos_axe, y-taille+y_pos_axe)

def deplacer_depart(x, y):
    deplacer(tortue, x, y)
    deplacer(point_rouge, x, y)
    point_rouge.dot(5, "red")
    point_rouge.write("   A", color="black", font=font)

# Grille
t = turtle.Turtle()
t.speed(0)
t.hideturtle()
t.width(1)
t.color('yellow')

def create_line(x0, y0, x1, y1):
    t.penup()
    t.goto(x0, y0)
    t.pendown()
    t.goto(x1, y1)

i = -taille
while i <= taille:
    create_line(-taille, i, taille, i)
    create_line(i, -taille, i, taille)
    i = i + 10

# Repere
t.color('black')
# Axes
create_line(-taille, -taille + y_pos_axe, taille, -taille + y_pos_axe)
create_line(-taille + x_pos_axe, -taille, -taille + x_pos_axe, taille)
# Tirets
for i in range(-1, 4):
    create_line(i*100 - 100, -5 - 100, i*100 - 100, 5 - 100)
    create_line(-5 - 100, i*100 - 100, 5 - 100, i*100 - 100)
# Nom des axes et échelle
t.penup()
deplacer(t, 380, 5)
t.write("x")
deplacer(t, 5, 380)
t.write("y")
deplacer(t, 85, -25)
t.write("100")
deplacer(t, -45, 95)
t.write("100")

# Mise en place des tortues
tortue = turtle.Turtle()
tortue.shape("turtle")
tortue.penup()
tortue.speed(0)

point_rouge = turtle.Turtle()
point_rouge.pencolor("red")
point_rouge.penup()
point_rouge.speed(0)
point_rouge.hideturtle()

xT, yT = 10, 20
deplacer_depart(xT, yT)

# les 4 cibles
coords_cibles = (250,  390), \
                (250,  200), \
                (250,   20), \
                (250, -100)

drawer = turtle.Turtle()
for i, coords in enumerate(coords_cibles):
    x, y = coords
    drawer.penup()
    drawer.speed(0)
    deplacer(drawer, x+15, y-10)
    drawer.write("BCDE"[i])
    deplacer(drawer, x, y)
    drawer.dot(10, "lightgreen")
    drawer.dot(1, "black")
deplacer(drawer, -1000, 0)

# Vitesse de 1 à 50, 0 pour le plus vite possible
tortue.speed(8)
point_rouge.speed(0)
tortue.pendown()
point_rouge.pendown()

turtle.done()

# Valeurs à changer
Δx = 10
Δy = 10

# Déplacement
while xT < x_max:
    xT = xT + Δx
    deplacer(tortue, xT, yT)
    yT = yT + Δy
    deplacer(tortue, xT, yT)

# Tracé de la ligne rouge
deplacer(point_rouge, xT, yT)

Exercice

Pour chacune des quatre cibles :

  • déterminer le coefficient directeur de la droite passant par la cible et le point rouge ;
  • faire apparaître sur l'écran une première ligne rouge - puis une deuxième - reliant le point rouge à la cible en exécutant le programme avec des valeurs bien choisies des variables Δx et Δy (noter les valeurs qui fonctionnent sur le cahier de recherche).
